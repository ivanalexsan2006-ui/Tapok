// ===== ПОЛНОСТЬЮ ЗАМЕНИТЬ функцию subscribeToPush ===== async function subscribeToPush() { console.log('📱 Проверка поддержки уведомлений...'); // Проверка поддержки браузером if (!('Notification' in window)) { console.log('❌ Браузер не поддерживает уведомления'); return; } if (!('serviceWorker' in navigator)) { console.log('❌ Service Worker не поддерживается'); return; } // Определяем iOS const isIOS = /iPad|iPhone|iPod/.test(navigator.userAgent); const isSafari = /^((?!chrome|android).)*safari/i.test(navigator.userAgent); console.log(`📱 Устройство: ${isIOS ? 'iOS' : 'другое'}, Safari: ${isSafari}`); try { // Запрашиваем разрешение console.log('🔔 Запрашиваем разрешение на уведомления...'); const permission = await Notification.requestPermission(); console.log('📱 Разрешение:', permission); if (permission === 'granted') { // Регистрируем Service Worker console.log('🔄 Регистрация Service Worker...'); // Удаляем старый Service Worker если есть const registrations = await navigator.serviceWorker.getRegistrations(); for (let registration of registrations) { await registration.unregister(); console.log('🗑️ Удален старый Service Worker'); } // Регистрируем новый const registration = await navigator.serviceWorker.register('/service-worker.js', { scope: '/' }); console.log('✅ Service Worker зарегистрирован:', registration); // Ждем активации await navigator.serviceWorker.ready; console.log('✅ Service Worker активен'); // Проверяем поддержку push if (!registration.pushManager) { console.log('❌ PushManager не поддерживается'); return; } // Получаем существующую подписку или создаем новую let subscription = await registration.pushManager.getSubscription(); if (subscription) { console.log('📦 Найдена существующая подписка'); // Проверяем, работает ли подписка try { const testRes = await fetch('/api/push/status', { headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}` } }); const testData = await testRes.json(); if (!testData.subscribed) { // Подписка не работает - удаляем await subscription.unsubscribe(); subscription = null; console.log('🗑️ Удалена нерабочая подписка'); } } catch (e) { console.log('❌ Ошибка проверки подписки'); } } if (!subscription) { console.log('🆕 Создаем новую подписку...'); // Конвертируем ключ const base64Key = 'BEl62iUYgUivxIkv69yViEuiBIa-Ib9-SkvMeAtA3LFgDzkrxZJjSgSnfckjBJuBkr3qBUYIHBQFLXYp5Nksh8U'; const key = urlBase64ToUint8Array(base64Key); // Создаем подписку subscription = await registration.pushManager.subscribe({ userVisibleOnly: true, applicationServerKey: key }); console.log('✅ Подписка создана'); } // Отправляем подписку на сервер console.log('📤 Отправляем подписку на сервер...'); const res = await fetch('/api/push/subscribe', { method: 'POST', headers: { 'Authorization': `Bearer ${localStorage.getItem('token')}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ subscription }) }); if (res.ok) { console.log('✅ Подписка сохранена на сервере'); // Для iOS показываем инструкцию if (isIOS) { showIOSInstructions(); } } else { console.log('❌ Ошибка сохранения подписки'); } } } catch (e) { console.error('❌ Ошибка подписки на уведомления:', e); } } // Инструкция для iOS function showIOSInstructions() { const instructions = document.createElement('div'); instructions.style.cssText = ` position: fixed; bottom: 20px; left: 20px; right: 20px; background: #2a6ee0; color: white; padding: 15px; border-radius: 10px; z-index: 1000; box-shadow: 0 4px 10px rgba(0,0,0,0.3); font-size: 14px; text-align: center; `; instructions.innerHTML = ` 📱 Для уведомлений на iOS:
1. Нажмите "Поделиться" ⎙
2. Выберите "На экран «Домой»"
3. Открывайте приложение с экрана `; document.body.appendChild(instructions); }
